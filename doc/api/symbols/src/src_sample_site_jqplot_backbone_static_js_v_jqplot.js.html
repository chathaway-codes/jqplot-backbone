1 define(["jquery", "backbone", "underscore", "backbone-tastypie", "jquery.jqplot",
  2         "plugins/jqplot.canvasAxisTickRenderer.min", "plugins/jqplot.canvasTextRenderer.min", "plugins/jqplot.categoryAxisRenderer.min",
  3     ], function($, Backbone, _) {
  4     /** 
  5      * jqplot class
  6      * <p>This class provides an API for generating graphs.
  7      *  It handles registering itself with the Backbone collection
  8      *  so that it can render as soon as the collection is fetched.</p>
  9      * @name jqplot
 10      * @class jqplot
 11      * @constructor
 12      * @param options
 13      *  The options parameter should be a hash. It will accept the following options:
 14      * <ul>
 15      *   <li>jqplotOptions - These options will be merged with the default options provided by this class.
 16      *     Preference will be given to the options provided in this constructor.</li>
 17      *   <li>variables - An array of the variables that should be used for constructing the graphs.
 18      *     These variables will fetched from each model provided by the colleciton and inserted 
 19      *      into an array that will be given to jqplot in a manner specified by the variableType option.
 20      *   </li>
 21      *   <li>variableType - This indicates how the variables will be put into an array. The valid options are:
 22      *    <ul>
 23      *      <li>groups - Data is "grouped" in a way that can be used for bar charts. Each variable represents one
 24      *        bar in each group.</li>
 25      *      <li>lines - Data is out into tuples consisting of ["Tick", "Value]. This data representation is appropiate
 26      *        line charts.</li>
 27      *      <li>pie - Data is put into a form appropiate for pie charts. Each variable is summed across the collection,
 28      *        and put into a single slice in the pie chart</li>
 29      *      <li>function(collection) - This should be a function that accepts a variable and returns an array of data.
 30      *        The data will be provided to this function in the first parameter, as the fetched collection. Please look
 31      *          at the jqplot documentation for how this data should be parsed.</li>
 32      *    </ul>
 33      *   <li>collection - The collection that will be used as the data source for this chart. This class will register itself
 34      *     with the collection so that it calls the "render" function when the data is available. This allows for async data requests,
 35      *     and the reuse of collections. The following things may be passed in as the variable:
 36      *      <ul>
 37      *          <li>A single collection</li>
 38      *          <li>An array of data to use for the graph. This will just be used as is.</li>
 39      *          <li>An array of collection. You MUST provide the matching number of "variable" arrays for this to work.
 40      *              Also note that you should NOT call fetch on a single collection multiple times before this renders,
 41      *              or it may break.</li>
 42      *   </li>
 43      *   <li>ticks - This variable will be used to generate the "ticks" for the line charts. It is passed into jqplot as
 44      *     the "['axes']['xaxis']['ticks']" option. This should be either null, or an array of strings.</li>
 45      * </ul>
 46      * @return jqplot view object
 47      */
 48     "use strict";

 50     /**
 51      * Takes two hashes and merges them, returning the merged hash
 52      *  That is, it goes through all elements in the array and
 53      *   combines hashes, otherwise prefering the second hash.
 54      * @param hash1 The "default" hash
 55      * @param hash2 The "preferred" hash
 56      */
 57     function mergeHash(hash1, hash2) {
 58         var out = {};
 59         for(var key in hash2) {
 60             if(hash2[key] != null) {
 61                 if(typeof hash2[key] == "object" && hash1[key] != null) {
 62                     // Recurse and merge
 63                     out[key] = mergeHash(hash1[key], hash2[key]);
 64                 } else {
 65                     // Else, take the specified value over the default
 66                     out[key] = hash2[key];
 67                 }
 68             }
 69         }
 70 
 71         // And make sure to keep the keys from hash1
 72         for(var key in hash1) {
 73             if(out[key] == null) {
 74                 out[key] = hash1[key];
 75             }
 76         }
 77         return out;
 78     }

 80     var view = Backbone.View.extend({
 81         jqplotOptions: {
 82             legend: { show:true, location: 'ne' },
 83             axesDefaults: {
 84                 tickRenderer: $.jqplot.CanvasAxisTickRenderer ,
 85                 tickOptions: {
 86                     angle: 30,
 87                 },
 88             },
 89             axes: {
 90               xaxis: {
 91                 renderer: $.jqplot.CategoryAxisRenderer
 92               },
 93             }
 94         },
 95         /**
 96          * This method is responsible for initializing all the variables,
 97          *  scanning through the supplied options and meshing them with our defaults,
 98          *  and setting anything that needs to be set
 99          * You must call fetch for the graphs to render!
100          * You can specify the following in the constructor:
101          *  -> jqplotOptions
102          *  -> variables
103          *  -> variableType ["groups", "lines", "pie", function(collection)]
104          *  -> collection
105          *  -> ticks
106          */
107         initialize: function() {
108             if(this.options.jqplotOptions != null) {
109                 this.jqplotOptions = mergeHash(this.jqplotOptions, this.options.jqplotOptions);
110             }
111             // Make sure a collection is specified
112             if(this.options.collection == null) {
113                 throw "A collection must be specified to query against!";
114             }
115             // Make sure both X and Y values are specified
116             if(this.options.variables == null && !_.isFunction(this.options.variableType)) {
117                 throw "An array of variables is required.";
118             }
119 
120             if(this.options.variableType != null) {
121                 this.variableType = this.options.variableType;
122             }
123 
124             this.variables = this.options.variables;
125             if(this.options.ticks != null)
126                 this.ticks = this.options.ticks;
127 
128             if(this.collection instanceof Array) {
129                 // If this is an array of collection, bind to all of them
130                 if(this.collection[0] instanceof Backbone.Collection) {
131                     for(var i=0; i < this.collection.length; i++) {
132                         this.collection[i].bind('reset', this.render, this);
133                     }
134                 } else {
135                     // Else, put it into a collection and call render
136                     var self = this;
137                     setTimeout(function() {
138                         self.collection = new Backbone.Collection(self.collection);
139                         self.render();
140                     }, 1);
141                 }
142             } else {
143                 this.collection.bind('reset', this.render, this);
144             }
145 
146             self = this;
147         },
148 
149 
150         /**
151          * This is called after the collection has been fetched
152          */
153         render: function() {
154             // Create the list that will be fed into jqplot
155             var elements;
156 
157             if(this.variableType == "groups") {
158                 elements = this.dataGroups();
159             } else if(this.variableType == "lines") {
160                 elements = this.dataLines();
161             } else if(this.variableType == "pie") {
162                 elements = this.dataPie();
163             } else if(_.isFunction(this.variableType)) {
164                 elements = this.variableType(this.collection);
165             } else {
166                 throw "Unknown variable type: " + this.variableType;
167             }
168 
169             // Get the ticks, if they are needed
170             if(this.ticks != null) {
171                 if(_.isFunction(this.ticks))
172                     this.jqplotOptions['axes']['xaxis']['ticks'] = this.ticks(this.collection);
173                 else
174                     this.jqplotOptions['axes']['xaxis']['ticks'] = this.ticks;
175             }
176 
177             if(elements == null || elements.length == 0 || elements[0].length == 0)
178                 return;
179 
180             $.jqplot($(this.el).attr('id'), elements, this.jqplotOptions);
181         },
182 
183         dataGroups: function() {
184             var self = this;
185             var elements = [];
186             this.collection.forEach(function(e) {
187                 for(var i=0; i < self.variables.length; i++) {
188                     if(elements[i] == null)
189                         elements.push([]);
190                     elements[i].push(e.get(self.variables[i]));
191                 }
192             });
193             return elements;
194         },
195 
196         dataLines: function() {
197             var self = this;
198             function addElements(e, variables) {
199                 var arr = []
200 
201                 for(var i=0; i < variables.length; i++) {
202                     arr.push(e.get(variables[i]));
203                 }
204                 return arr;
205             }
206 
207             function doAdd(collection, variables) {
208                 var elements = [];
209                 // If we are dealing with an array of variable variables, do this for each
210                 if(variables[0] instanceof Array) {
211                     for(var i=0; i < variables.length; i++) {
212                         var t_el = [];
213                         collection.forEach(function(e) {
214                             t_el.push(addElements(e, variables[i]));
215                         });
216                         elements.push(t_el);
217                     }
218                 }
219                 else {
220                     collection.forEach(function(e) {
221                         elements.push(addElements(e, variables));
222                     });
223     
224                     elements = [elements];
225                 }
226 
227                 return elements;
228             }
229 
230             // If we are dealing with an array of collection, recurse on each
231             if(this.collection instanceof Array) {
232                 // If there aren't enough variables to go with the collections, complain
233                 if(this.variables.length != this.collection.length)
234                     throw "Not enough variables given for the specified collection!";
235                 // First, make sure everything has loaded
236                 this.succeeded += 1;
237                 if(this.succeeded < this.collection.length)
238                     return;
239                 var elements = [];
240                 for(var i=0; i < this.collection.length; i++) {
241                     var arr = doAdd(this.collection[i], this.variables[i])
242                     elements.push(arr[0]);
243                  }
244                  this.succeeded = 0;
245                  return elements;
246              } else {
247                 return doAdd(this.collection, this.variables);
248              }
249     
250 
251         },
252 
253         dataPie: function() {
254             var self = this;
255             var elements = [];
256 
257             this.collection.forEach(function(e) {
258                 for(var i=0; i < self.variables.length; i++) {
259                     if(elements[i] == null) {
260                         elements.push([self.variables[i], 0]);
261                         // If there is a label specified, use that
262                         if(self.ticks[i] != null)
263                             elements[i][0] = self.ticks[i];
264                     }
265                     elements[i][1] = elements[i][1] + e.get(self.variables[i]);
266                 }
267             });
268 
269             return [elements];
270         },
271 
272         /**
273          * This function calls the function specified by the "failure" element in the prototype
274          */
275         fetchFailure: function(collection, response, options) {
276             // This is bad. Call the user specified failure function? Or throw?
277             if(this.failure != null)
278                 this.failure(response);
279             else
280                 throw "Failed to fetch data: " + response;
281         },
282 
283         variableType: "lines",
284         // This is used to count how many collections have been loaded
285         succeeded: 0,
286     });
287 
288     return view;
289 });
