1 define(["jquery", "backbone", "underscore", "backbone-tastypie", "jquery.jqplot"], function($, Backbone, _) {
  2     /** 
  3      * jqplot class
  4      * <p>This class provides an API for generating graphs.
  5      *  It handles registering itself with the Backbone collection
  6      *  so that it can render as soon as the collection is fetched.</p>
  7      * @name jqplot
  8      * @class jqplot
  9      * @constructor
 10      * @param options
 11      *  The options parameter should be a hash. It will accept the following options:
 12      * <ul>
 13      *   <li>jqplotOptions - These options will be merged with the default options provided by this class.
 14      *     Preference will be given to the options provided in this constructor.</li>
 15      *   <li>variables - An array of the variables that should be used for constructing the graphs.
 16      *     These variables will fetched from each model provided by the colleciton and inserted 
 17      *      into an array that will be given to jqplot in a manner specified by the variableType option.
 18      *   </li>
 19      *   <li>variableType - This indicates how the variables will be put into an array. The valid options are:
 20      *    <ul>
 21      *      <li>groups - Data is "grouped" in a way that can be used for bar charts. Each variable represents one
 22      *        bar in each group.</li>
 23      *      <li>lines - Data is out into tuples consisting of ["Tick", "Value]. This data representation is appropiate
 24      *        line charts.</li>
 25      *      <li>pie - Data is put into a form appropiate for pie charts. Each variable is summed across the collection,
 26      *        and put into a single slice in the pie chart</li>
 27      *      <li>function(collection) - This should be a function that accepts a variable and returns an array of data.
 28      *        The data will be provided to this function in the first parameter, as the fetched collection. Please look
 29      *          at the jqplot documentation for how this data should be parsed.</li>
 30      *    </ul>
 31      *   <li>collection - The collection that will be used as the data source for this chart. This class will register itself
 32      *     with the collection so that it calls the "render" function when the data is available. This allows for async data requests,
 33      *     and the reuse of collections.</li>
 34      *   <li>ticks - This variable will be used to generate the "ticks" for the line charts. It is passed into jqplot as
 35      *     the "['axes']['xaxis']['ticks']" option. This should be either null, or an array of strings.</li>
 36      * </ul>
 37      * @return jqplot view object
 38      */
 39     "use strict";

 41     /**
 42      * Takes two hashes and merges them, returning the merged hash
 43      *  That is, it goes through all elements in the array and
 44      *   combines hashes, otherwise prefering the second hash.
 45      * @param hash1 The "default" hash
 46      * @param hash2 The "preferred" hash
 47      */
 48     function mergeHash(hash1, hash2) {
 49         var out = {};
 50         for(var key in hash2) {
 51             if(hash2[key] != null) {
 52                 if(typeof hash2[key] == "object" && hash1[key] != null) {
 53                     // Recurse and merge
 54                     out[key] = mergeHash(hash1[key], hash2[key]);
 55                 } else {
 56                     // Else, take the specified value over the default
 57                     out[key] = hash2[key];
 58                 }
 59             }
 60         }
 61 
 62         // And make sure to keep the keys from hash1
 63         for(var key in hash1) {
 64             if(out[key] == null) {
 65                 out[key] = hash1[key];
 66             }
 67         }
 68         return out;
 69     }

 71     var view = Backbone.View.extend({
 72         jqplotOptions: {
 73             axesDefaults: {
 74                 tickRenderer: $.jqplot.CanvasAxisTickRenderer ,
 75                 tickOptions: {
 76                     angle: 30,
 77                 },
 78             },
 79             axes: {
 80               xaxis: {
 81                 renderer: $.jqplot.CategoryAxisRenderer
 82               },
 83             }
 84         },
 85         /**
 86          * This method is responsible for initializing all the variables,
 87          *  scanning through the supplied options and meshing them with our defaults,
 88          *  and setting anything that needs to be set
 89          * You must call fetch for the graphs to render!
 90          * You can specify the following in the constructor:
 91          *  -> jqplotOptions
 92          *  -> variables
 93          *  -> variableType ["groups", "lines", "pie", function(collection)]
 94          *  -> collection
 95          *  -> ticks
 96          */
 97         initialize: function() {
 98             if(this.options.jqplotOptions != null) {
 99                 this.jqplotOptions = mergeHash(this.jqplotOptions, this.options.jqplotOptions);
100             }
101             // Make sure a collection is specified
102             if(this.options.collection == null) {
103                 throw "A collection must be specified to query against!";
104             }
105             // Make sure both X and Y values are specified
106             if(this.options.variables == null && !_.isFunction(this.options.variableType)) {
107                 throw "An array of variables is required.";
108             }
109 
110             if(this.options.variableType != null) {
111                 this.variableType = this.options.variableType;
112             }
113 
114             this.variables = this.options.variables;
115             if(this.options.ticks != null)
116                 this.ticks = this.options.ticks;
117 
118             this.collection.bind('reset', this.render, this);
119 
120             self = this;
121 
122             this.failure = this.options.failure;
123             //this.options.collection.fetch({success: this.fetchComplete, failure: this.fetchFailure});
124         },
125 
126 
127         /**
128          * This is called after the collection has been fetched
129          */
130         render: function() {
131             // Create the list that will be fed into jqplot
132             var elements;
133 
134             if(this.variableType == "groups") {
135                 elements = this.dataGroups();
136             } else if(this.variableType == "lines") {
137                 elements = this.dataLines();
138             } else if(this.variableType == "pie") {
139                 elements = this.dataPie();
140             } else if(_.isFunction(this.variableType)) {
141                 elements = this.variableType(this.collection);
142             } else {
143                 throw "Unknown variable type: " + this.variableType;
144             }
145 
146             // Get the ticks, if they are needed
147             console.log(this.ticks);
148             if(this.ticks != null) {
149                 if(_.isFunction(this.ticks))
150                     this.jqplotOptions['axes']['xaxis']['ticks'] = this.ticks(this.collection);
151                 else
152                     this.jqplotOptions['axes']['xaxis']['ticks'] = this.ticks;
153                 console.log(this.jqplotOptions['axes']['xaxis']['ticks']);
154             }
155 
156             $.jqplot($(this.el).attr('id'), elements, this.jqplotOptions);
157         },
158 
159         dataGroups: function() {
160             var self = this;
161             var elements = [];
162             this.collection.forEach(function(e) {
163                 for(var i=0; i < self.variables.length; i++) {
164                     if(elements[i] == null)
165                         elements.push([]);
166                     elements[i].push(e.get(self.variables[i]));
167                 }
168             });
169             return elements;
170         },
171 
172         dataLines: function() {
173             var elements = []
174             var self = this;
175             this.collection.forEach(function(e) {
176                 var elements_inner = [];
177                 for(var i=0; i < self.variables.length; i++) {
178                     elements_inner.push(e.get(self.variables[i]));
179                 }
180                 elements.push(elements_inner);
181             });
182             return [elements];
183         },
184 
185         dataPie: function() {
186             var self = this;
187             var elements = [];
188 
189             this.collection.forEach(function(e) {
190                 for(var i=0; i < self.variables.length; i++) {
191                     if(elements[i] == null) {
192                         elements.push([self.variables[i], 0]);
193                         // If there is a label specified, use that
194                         if(self.ticks[i] != null)
195                             elements[i][0] = self.ticks[i];
196                     }
197                     elements[i][1] = elements[i][1] + e.get(self.variables[i]);
198                 }
199             });
200 
201             return [elements];
202         },
203 
204         /**
205          * This function calls the function specified by the "failure" element in the prototype
206          */
207         fetchFailure: function(collection, response, options) {
208             // This is bad. Call the user specified failure function? Or throw?
209             if(this.failure != null)
210                 this.failure(response);
211             else
212                 throw "Failed to fetch data: " + response;
213         },
214 
215         variableType: "lines",
216     });
217 
218     return view;
219 });
